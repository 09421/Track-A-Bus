\subsection{Use Case 4: Tilgå log}
\textbf{Mål:}
\\
Målet med denne Use Case er at tilgå en logfil og få denne vist.
\\
\\
\textbf{Initiering:}
\\
Use Casen initieres ved at operatøren/programmøren\footnote{Da det både er programmøren og operatøren der kan anvende Use Casen betegnes de samlet som bruger.} tilkendegiver overfor systemet, at han vil tilgå logfiler. 
\\
\\
\textbf{Aktører og interessenter:}
\\
Primære aktører:
	\begin{itemize}
	\item Operatør.
	\item Programmør.
	\end{itemize}
Sekundære aktører:
	\begin{itemize}
	\item Databasen.
	\end{itemize}
\textbf{Antal samtidige forekomster:}
\\
En samtidig forekomst.
\\\\
\textbf{Ikke funktionelle krav:}
\begin{itemize}
\item Loggen udskrives med tid [hh.mm.ss] eks. [15:02:42].
\end{itemize}
\textbf{Referencer:}
\\
Ingen.
\\\\
\textbf{Startbetingelser:}
\\
Der skal være forbindelse til databasen, så logfiler kan hentes.
\\\\
\textbf{Slutresultat ved succes:}
\\
Logfiler er blevet hentet og gemt.
\\\\
\textbf{Slutresultat ved undtagelser:}
\\
Logfiler er ikke blevet hentet eller gemt.
\\\\
\textbf{Normalforløb}
	\begin{enumerate}
	\item Brugeren tilkendegiver overfor systemet, at denne ønsker at hente en logfil.
	\item Systemet lister alle tilgængelige logfiler.
	\item Brugeren tilkendegiver overfor systemet, hvilken logfil der skal vises.
	\item Systemet viser data for den givne logfil.
	\item Brugeren vælger at gemme logfilen.
	\item Systemet åbner et vindue, hvor brugeren kan vælge, hvor logfilen skal gemmes.
	\item Brugeren tilkendegiver overfor systemet, hvor logfilen skal gemmes, samt med hvilket navn den skal gemmes med.
	\item Systemet gemmer logfilen på computeren.
	\end{enumerate}
\textbf{Undtagelser}
\begin{enumerate}[\text{Undtagelse *} a:]
	\item Der mistes forbindelse til databasen.
		\begin{enumerate}[1.]
		\item Gennem en visuel alarm bliver brugeren informeret om, at forbindelsen til databasen er blevet afbrudt.
		\item Brugeren bliver spurgt, om han vil vente på, at forbindelsen bliver genetableret. 
		
		\begin{enumerate}
		\item Brugeren tilkendegiver overfor systemet, at dette ønskes.
			\begin{enumerate}
			\item Systemet forsøger at genetablere forbindelsen til databasen.
			\item Hvis forbindelsen ikke kan genoprettes, præsenteres brugeren for punkt 2.
			\end{enumerate}
		
		\item Brugeren tilkendegiver overfor systemet, at dette ikke ønskes.
			\begin{enumerate}
			\item Programmet lukker ned.
			\end{enumerate}
		\end{enumerate}
	\end{enumerate}
\end{enumerate}


	\begin{enumerate}[\text{Undtagelse 2} a:]
	\item Der er ikke nogle logfiler i listen.
	
		\begin{enumerate}
		\item Systemet viser ingen data fra logfilerne.
		\end{enumerate}
	\end{enumerate}

	\begin{enumerate}[\text{Undtagelse 5} a:]
		\item Brugeren vil ikke gemme logfilen, men blot have den vist.
		
		\begin{enumerate}
		\item Brugeren tilkendegiver ikke overfor systemet, at han vil gemme logfilen.
		\end{enumerate}
	\end{enumerate}
	
	
	\begin{enumerate}[\text{Undtagelse 6} a:]
	\item Brugeren tilkendegiver overfor systemet, at han ikke ønsker at gemme filen alligevel.
		
		\begin{enumerate}
		\item Systemet lukker vinduet til, at gemme placering for filen.
		\end{enumerate}
	\end{enumerate}
		
	\begin{enumerate}[\text{Undtagelse 7} a:]
	\item Filnavnet er allerede optaget.
		
		\begin{enumerate}
		\item Systemet spørger brugeren om denne ønsker at overskrive den eksisterende fil.
			
			\begin{enumerate}
			\item Brugeren tilkendegiver overfor systemet at dette ønskes.
				
				\begin{enumerate}
				\item Filen overskrives med den nye.
				\end{enumerate}
			
				\item Brugeren tilkendegiver overfor systemet, at dette ikke ønskes.
				
				\begin{enumerate}
				\item Brugeren får lov til at indtaste et nyt navn på filen.
				\end{enumerate}
			
			\end{enumerate}
		
		\end{enumerate}
	
	\end{enumerate}
