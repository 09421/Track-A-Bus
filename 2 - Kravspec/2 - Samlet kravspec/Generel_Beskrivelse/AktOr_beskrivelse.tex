\subsubsection{Aktør beskrivelse}

\underline{\textbf{Primære aktører:}}\\
\begin{tabular}{lp{10 cm}}

\textbf{Aktør navn} 				& Operatør\\
\textbf{Beskrivelse} 				& Denne aktør starter og stopper systemet. Målet for aktøren er at få sorteret klodser, alt efter deres materialetype. \\
\textbf{Antal samtidige aktører}	& 1 \\
\\\\
\textbf{Aktør navn}					&  Programmør. \\ 
\textbf{Beskrivelse} 				&  Denne aktørs opgave er at lave brugerdefinerede programmer til systemet.\\ 
\textbf{Antal samtidige aktører} 	&  1
\end{tabular}
\\\\
\\\\
\underline{\textbf{Sekundære aktører}}
\\
\begin{tabular}{lp{10 cm}}

\textbf{Aktør navn} 				& Vægt\\
\textbf{Beskrivelse} 				& Denne aktør vejer et objekt, så systemet kan bestemme dets materialetype.  \\
\textbf{Antal samtidige aktører}	& 1 \\
\\\\
\textbf{Aktør navn}					& Transportbåndssensor \\
\textbf{Beskrivelse}				& Denne aktør registrerer, hvornår et objekt er klar til at blive samlet op af robotten.  \\
\textbf{Antal samtidige aktører}	& 1 \\
\\\\
\textbf{Aktør navn} 				&  Robotarm.\\ 
\textbf{Beskrivelse} 				&  Denne aktørs opgave er at flytte og måle forskellige typer klodser.\\ 
\textbf{Antal samtidige aktører} 	&  1 \\ 
\\\\
\textbf{Aktør navn} 				&  Database. \\ 
\textbf{Beskrivelse} 				&  Databasen indeholder logfiler for systemets data for de enkelte klodser, samt informationer om robotten og systemets status. Ligeledes indeholder databasen programmerne, som programmøren har lavet.\\ 
\textbf{Antal samtidige aktører} 	&  1
\end{tabular}
