\subsubsection{Use Case 1}
\textbf{Test Case: Normalforløb}
\\\begin{tabular}{l p{11cm}}
FORBEREDELSE:	&	Robotten skal være sat til at køre standardprogrammet. Ligeledes skal robotten være i startposition. Der skal være placeret en grøn klods på transportbåndet (5*5*5 CM, 200g - Densitet = 1,665), desuden skal databasen for sorterede klodser være tom.
\\
BESKRIVELSE:	&	Der testes, at det er muligt at sortere en klods i forhold til dens densitet.  
\\\\
\end{tabular}
\\
\begin{testcase}
\punkt 1
\Aktion Operatøren tilkendegiver overfor systemet, at sorteringen skal starte, ved klik på startknappen.
\Forventet Transportbåndet startes

\punkt 2
\Aktion Den grønne klods kører ind foran sensoren.
\Forventet Transportbåndet stoppes

\punkt 3
\Aktion Klodsen måles og vejes (se acceptest for \textit{Include 1.1 Mål og vej klods})
\Forventet Klodsens mål er bestemt til 5x5x5CM ($\pm$ 0,5 CM) og vægt er bestemt til 200g ($\pm$ 5 g) (se acceptest for \textit{Include 1.1 Mål og vej klods})

\punkt 4
\Aktion Klodsens materialetype bestemmes(se acceptest for \textit{Include 1.2 Mål og vej klods})
\Forventet Klodsens materialetype er bestemt til grøn (se acceptest for \textit{Include 1.2 Mål og vej klods})
\end{testcase}
\begin{testcase}
\punkt 5
\Aktion -||-
\Forventet Klodsen er placeret i rum 1, hvori grønne klodser tilhører.

\punkt 6.1
\Aktion -||-
\Forventet Robotarmen er ført tilbage til startposition

\punkt 6.2
\Aktion Hele normal forløbet er kørt igennem.
\Forventet Ved hver aktion skrives en log-besked til loggingvinduet, der specificerer, hvad der er set.
\end{testcase}

\textbf{Test Case: afvigelsesforløb 5a}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:	&	Rummet til de grønne klodser skal være fuldt. Dette skal være registeret i databasen, og der skal være placeret en grøn klods på transprotbåndet.\\

\begin{testcase}

\textbf{Test Case: Ikke-funktionelle krav 1}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:	&	Systemet skal være sat til at køre standardprogrammet.  \\
BESKRIVELSE:	&	Der testes om det tager maksimum tre minutter at sortere en klods alt efter materialetype.  \\ 
\end{tabular}
\\\\
\begin{testcase}
\punkt 1
\Aktion Sorteringsprogrammet startes samtidig med, at et stopur startes. Når klodsen er sorteres, stoppes uret.
\Forventet Sorteringsprogrammet tager mindre end tre minutter.
\end{testcase}
\\
\textbf{Test Case: Ikke-funktionelle krav 2}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:	&	Systemet skal være sat til at køre standardprogrammet. Databasen må ikke indeholde data for nogen sorterede klodser.  \\
BESKRIVELSE:	&	Det testes om klodsen placeres med en maksimum afvigelse på 2 cm fra en ønskede koordinat, når den placeres i kassen indeholdende de sorterede klodser.   \\
\end{tabular}
\\\\
\begin{testcase}
\punkt 1
\Aktion Når programmet har kørt til ende, måles afstanden fra kassens ende til klodsen.
\Forventet Afstanden fra klodsen til kassens kant er mindre end 2 cm.
\end{testcase}
