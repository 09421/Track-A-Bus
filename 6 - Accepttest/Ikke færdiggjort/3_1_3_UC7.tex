\subsubsection{Use Case 7: Rediger busruteplan}
\textbf{Test Case: Normalforløb A-1}
\\
\begin{tabular}{ l  p{12cm} }
FORBEREDELSE: & Administratoren er logget ind på administrations hjemmesiden, samt står på 'Rediger busrute' siden.
\\
BESKRIVELSE: & Det testes at det er muligt at tilføje ny busrute til systemet.
\end{tabular}
\\
\begin{testcase}
\punkt 1
\Aktion Administratoren vælger minimum et stoppested fra listen, fra listen over alle stoppesteder i systemet, samt trykker på knappen 'Set bus stops'.
\Forventet De valgte stoppesteder kan ses indtegnet på kortet.
\punkt 2
\Aktion Administratoren trykker på kortet, hvor han ønsker busruten skal starte, hvorefter han trykker på kortet, hvor han ønsker busruten skal ende.
\Forventet En busrute vil blive indtegnet på kortet, med start og slut punkt der passer overens med der hvor der blev trykket på kortet.
\punkt 3
\Aktion Administratoren indtaster navnet "testruteafskåret" og tilkendegiver overfor systemet at han ønsker ruten persisteret, ved tryk på 'Save' knappen.
\Forventet ruten "testruteaf (SE IFK-A)" bliver persisteret på databasen.
\punkt IFK-A 
\Aktion Siden refreshes efter punkt 3.
\Forventet Rutenavnet er sat til "testruteaf"
\punkt IFK-B
\Aktion databasen åbnes efter punkt 3.
\Forventet Alle punkter tastet ind med minimum seks decimaler.
\end{testcase}

\textbf{Test Case: Normalforløb B}
\\
\begin{tabular}{ l  p{12cm} }
FORBEREDELSE: & Administratoren er logget ind på administrations hjemmesiden, samt står på 'Rediger busrute' siden. Normalforløb A-1 skal være gennemført.
\\
BESKRIVELSE: & Det testes at det er muligt at ændre i en busrute fra systemet.
\end{tabular}
\\
\begin{testcase}
\punkt 1
\Aktion Administratoren vælger busrute "testruteaf" fra listen over ruter i systemet.
\Forventet Den valgte busrute bliver indteget på kortet.
\punkt 2
\Aktion Administratoren flytter punkt A til et nyt sted på kortet.
\Forventet Ruten ændres til at starte fra det sted, punkt A er flyttet til.
\punkt 3
\Aktion Administratoren tilkendegiver overfor systemet at han ønsker ændringer persisteret, ved tryk på 'Save' knappen.
\Forventet ændringer bliver persisteret på databasen.
\punkt Unt-A
\Aktion Efter punkt 2 deaktiveres internettet på computeren, og save knappen trykkes.
\Forventet Administratoren bliver præsenteret for en fejlmeddelelse, der beskriver at det ikke er muligt at persiterer ændringer.
\end{testcase}
