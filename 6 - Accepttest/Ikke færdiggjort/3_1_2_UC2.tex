\subsubsection{Use Case 2: Vis placering af alle busser, og stoppesteder på valgt rute}
\textbf{Test Case: Normalforløb A-1}
\\\begin{tabular}{l p{11cm}} 
FORBEREDELSE:        &        TrackABus android mobil-applikationen skal være startet, samt stå ved listen af busruter, desuden skal der i databasen være gemt rutepunkter, busstoppesteder samt en bus, for en valgt busrute og simulatoren skal være startet.
\\
BESKRIVELSE:        &        Der testes, at det er muligt at få vist en rute, stoppesteder på ruten, samt en bus, hvis position opdateres.
\\\\ 
\end{tabular}
\\
\begin{testcase}
\punkt 1
\Aktion Brugeren tilkendegiver overfor systemet at han ønsker at få vist den simple busrute, ved at trykke på den simple busrute, i listen af alle busruter.
\Forventet Der vises et kort, med indtegnet busrute, stoppesteder samt en bus, hvis position bliver opdateret.
\end{testcase}
\\

\textbf{Test Case: Ikke-funktionelle krav A}
\\\begin{tabular}{l p{11cm}} 
FORBEREDELSE:        &        Den simple busrute skal være valgt. Ruten, stoppesteder samt en bus skal være indtegnet, desuden skal simultoren være startet på den simple busrute.
\\
BESKRIVELSE:        &        Der testes, at bussens position opdateres hver sekund, plus en afvigelse på højst op til 0.5 sekunder.
\\\\ 
\end{tabular}
\\
\begin{testcase}
\punkt 1
\Aktion Et stopur startes på samme tid med at bussens position opdateres.
\Forventet Det kan ses at bussens position opdateres hver sekunds, med en maks afvigelse på 0.5 sekunder.
\end{testcase}

\textbf{Test Case: Undtagelsesforløb A}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:        &         TrackABus android mobil-applikationen skal være startet, samt stå ved listen af busruter, desuden skal der ikke være forbindelse til databasen.\\
BESKRIVELSE:        &        Der testes, at der bliver vist en fejlmeddelelse hvis der ikke kan skabes forbindelse til databasen.  \\ 
\end{tabular}
\\\\
\begin{testcase}
\punkt 1
\Aktion Brugeren tilkendegiver overfor systemet at han ønsker at få vist en simple busrute, ved tryk på den simple busrute , i listen af alle busruter.
\Forventet En fejlmeddelse bliver vist på skærmen, der beskriver det ikke er muligt at indlæse busruten.
\end{testcase}
\\
\textbf{Test Case: Undtagelsesforløb B}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:        &         Indlæsning af den simple busrute er igangsat, samt der er gemt rutepunkter, busstoppesteder samt en bus, for busruten, i databasen\\
BESKRIVELSE:        &        Der testes, at hvis indlæsning af data bliver annulleret, ved tryk på telefonens tilbage-knap, returneres der til listen af busruter.  \\ 
\end{tabular}
\\\\
\begin{testcase}
\punkt 1
\Aktion brugeren annullerer indlæsning af den simple busrute.
\Forventet Indlæsning af busruten stoppes, samt der retuneres til listen af busruter.
\end{testcase}
\\
\textbf{Test Case: Undtagelsesforløb C}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:        &         Indlæsning af den simple rute er igangsat, samt der er gemt rutepunkter, busstoppesteder samt en bus, for busruten, i databasen\\
BESKRIVELSE:        &        Der testes, at hvis systemet går i dvale, ved tryk på telefonens home-knap,vil systemet indlæse den simple busrute ind fra databasen i baggrunden.  \\ 
\end{tabular}
\\\\
\begin{testcase}
\punkt 1
\Aktion Brugeren trykker på telefonens home-knap under indlæsning af den simple busrute.
\Forventet Systemet indlæser busruten færdig i baggrunden. Ved genåbning af programmet, vil busrute, stoppesteder samt bussen kunne ses indtegnet på et kort.
\end{testcase}
\\
\textbf{Test Case: Undtagelsesforløb D}
\\ \begin{tabular}{l p{11cm}}
FORBEREDELSE:        &         kortet vises, med indtegnet simpel busrute, stoppesteder samt en bus, hvis position opdateres\\
BESKRIVELSE:        &        Der testes, at hvis der mistes forbindelse til internetet imens systemet kører, vil der blive vist en fejlmeddelelse om at der ikke er forbindelse til internetet. Busserns position ikke længere vil blive opdateret.  \\ 
\end{tabular}

\begin{testcase}

\end{testcase}
